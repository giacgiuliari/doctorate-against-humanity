\begin{itemize}

    \item Ben draws a black card;
    \item Players select white cards from their hand to fill the blanks in the black card, and place them face down on the table;
    \item All the white cards are shuffled and revealed;
    \item Ben selects the best completion, and shares a memory/fun story about it;
    \item Players draw back to 10 white cards.
\end{itemize}
